    stage('Clean'){
        sh "rm -rf /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/*"
    }
retry(count: 2){    
    stage('clone frontend and devops'){

        dir('frontend'){
          git branch: 'equinor-frontend', credentialsId: 'dt-sachingithub97', url: 'https://github.com/PGuardians/em-dt-causal.git'
        }
        
          git branch: 'devops', credentialsId: 'dt-sachingithub97', url: 'https://github.com/PGuardians/em-dt-devops.git'
        
    }
    stage("Move Dependency files"){
        sh "cp -rf /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/dockerfile/twin-sim/frontend/Dockerfile  /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/Dockerfile"
        sh "cp -rf /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/kube-yml/twin-sim/equinor-frontend/frontend.yaml /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/frontend.yaml"
        sh "cp -rf /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/env/twin-sim/equinor-frontend/.env.local /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/frontend/.env.local"

    }
}
    
    
    stage("Docker Build and Push, Replace YML"){
        def previousbuild = currentBuild.previousSuccessfulBuild
        def REGISTRY = "privaterepo.com:5000"        
        def REPO = "twin-sim" 
        def JOB_NAME = "dt-equinor-frontend"

        if (previousbuild != null) {
             echo "Previous Successful Build Number: ${previousbuild.number}"
             def old_tag = "${JOB_NAME}-jen-v1.${previousbuild.number}"
             echo "Last Worked Image: ${REGISTRY}/${REPO}:${old_tag}"
        } else {
            echo "No previous successful build found."
        }
     
        def new_tag = "${JOB_NAME}-jen-v1.${currentBuild.number}"
        def IMAGE = "${REGISTRY}/${REPO}:${new_tag }"
        echo "Current Image: ${IMAGE}"


        

        sh "sed -i    's/RRR/${REGISTRY}/g' /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/frontend.yaml"
        sh "sed -i    's/rrr/${REPO}/g' /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/frontend.yaml"
        sh "sed -i    's/NNN/${new_tag}/g' /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/frontend.yaml"



        sh "docker build -t ${IMAGE} /home/eminds/jenkins/workspace/dt/dt-equinor-frontend/"

        sh "docker push ${IMAGE}"
    }
